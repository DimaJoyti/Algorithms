## 📊 Types of Firewalls

| Type | Layer | Description | Example |
|------|-------|-------------|---------|
| Packet Filtering | L3/L4 | Filters by IP, port, protocol | iptables, ACL |
| Stateful | L3/L4 | Tracks connection state | pfSense, Cisco ASA |
| Application | L7 | Inspects application data | WAF, Proxy |
| Next-Gen | L3-L7 | DPI, IPS, user identity | Palo Alto, Fortinet |

### Python Firewall Rule Generator

```python
from dataclasses import dataclass
from enum import Enum
from typing import Optional, List

class Action(Enum):
    ALLOW = "ACCEPT"
    DENY = "DROP"
    REJECT = "REJECT"

class Protocol(Enum):
    TCP = "tcp"
    UDP = "udp"
    ICMP = "icmp"
    ANY = "all"

@dataclass
class FirewallRule:
    action: Action
    source: str
    destination: str
    port: Optional[int]
    protocol: Protocol
    description: str = ""
    rate_limit: Optional[int] = None  # requests per second
    
    def to_iptables(self, chain: str = "INPUT") -> str:
        """Generate iptables command"""
        cmd = f"iptables -A {chain}"
        
        if self.source != "Any":
            cmd += f" -s {self.source}"
        
        if self.destination != "Any":
            cmd += f" -d {self.destination}"
        
        cmd += f" -p {self.protocol.value}"
        
        if self.port:
            cmd += f" --dport {self.port}"
        
        if self.rate_limit:
            cmd += f" -m limit --limit {self.rate_limit}/second"
        
        cmd += f" -j {self.action.value}"
        
        return cmd
    
    def to_cloud_format(self) -> dict:
        """Generate cloud firewall rule format"""
        return {
            "action": self.action.value,
            "source": self.source,
            "destination": self.destination,
            "port": self.port,
            "protocol": self.protocol.value,
            "description": self.description
        }

class FirewallPolicy:
    def __init__(self, default_action: Action = Action.DENY):
        self.rules: List[FirewallRule] = []
        self.default_action = default_action
    
    def add_rule(self, rule: FirewallRule):
        self.rules.append(rule)
        return self
    
    def generate_iptables_script(self) -> str:
        """Generate complete iptables script"""
        script = [
            "#!/bin/bash",
            "# Firewall rules generated by FirewallPolicy",
            "",
            "# Flush existing rules",
            "iptables -F",
            "iptables -X",
            "",
            "# Default policies",
            f"iptables -P INPUT {self.default_action.value}",
            "iptables -P FORWARD DROP",
            "iptables -P OUTPUT ACCEPT",
            "",
            "# Allow loopback",
            "iptables -A INPUT -i lo -j ACCEPT",
            "iptables -A OUTPUT -o lo -j ACCEPT",
            "",
            "# Allow established connections",
            "iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT",
            "",
            "# Custom rules",
        ]
        
        for i, rule in enumerate(self.rules, 1):
            script.append(f"# Rule {i}: {rule.description}")
            script.append(rule.to_iptables())
        
        script.extend([
            "",
            "# Logging for dropped packets",
            "iptables -A INPUT -m limit --limit 5/min -j LOG --log-prefix \"DROPPED: \"",
            "",
            "echo 'Firewall rules applied successfully'"
        ])
        
        return "\n".join(script)

# Usage
policy = FirewallPolicy(default_action=Action.DENY)

policy.add_rule(FirewallRule(
    action=Action.ALLOW,
    source="Any",
    destination="Any",
    port=80,
    protocol=Protocol.TCP,
    description="Allow HTTP traffic"
))

policy.add_rule(FirewallRule(
    action=Action.ALLOW,
    source="Any",
    destination="Any",
    port=443,
    protocol=Protocol.TCP,
    description="Allow HTTPS traffic"
))

policy.add_rule(FirewallRule(
    action=Action.ALLOW,
    source="10.0.0.0/8",
    destination="Any",
    port=22,
    protocol=Protocol.TCP,
    description="Allow SSH from internal network",
    rate_limit=10
))

print(policy.generate_iptables_script())
```

## 🔧 iptables Implementation

### Common iptables Commands

```bash
# List all rules
iptables -L -n -v

# Allow HTTP/HTTPS
iptables -A INPUT -p tcp --dport 80 -j ACCEPT
iptables -A INPUT -p tcp --dport 443 -j ACCEPT

# Allow SSH with rate limiting
iptables -A INPUT -p tcp --dport 22 -m limit --limit 3/min -j ACCEPT

# Block specific IP
iptables -A INPUT -s 192.168.1.100 -j DROP

# Port forwarding
iptables -t nat -A PREROUTING -p tcp --dport 80 -j REDIRECT --to-port 8080

# NAT for internal network
iptables -t nat -A POSTROUTING -s 10.0.0.0/24 -o eth0 -j MASQUERADE

# Save rules (Ubuntu/Debian)
iptables-save > /etc/iptables/rules.v4
```

### Go iptables Manager

```go
package main

import (
	"fmt"
	"os/exec"
	"strings"
)

type IPTablesManager struct {
	table string
}

func NewIPTablesManager(table string) *IPTablesManager {
	if table == "" {
		table = "filter"
	}
	return &IPTablesManager{table: table}
}

func (m *IPTablesManager) AddRule(chain, source, dest, protocol, port, action string) error {
	args := []string{"-t", m.table, "-A", chain}
	
	if source != "" {
		args = append(args, "-s", source)
	}
	if dest != "" {
		args = append(args, "-d", dest)
	}
	if protocol != "" {
		args = append(args, "-p", protocol)
	}
	if port != "" {
		args = append(args, "--dport", port)
	}
	args = append(args, "-j", strings.ToUpper(action))
	
	cmd := exec.Command("iptables", args...)
	output, err := cmd.CombinedOutput()
	if err != nil {
		return fmt.Errorf("iptables error: %s", output)
	}
	return nil
}

func (m *IPTablesManager) ListRules(chain string) ([]string, error) {
	args := []string{"-t", m.table, "-L", chain, "-n", "--line-numbers"}
	cmd := exec.Command("iptables", args...)
	output, err := cmd.Output()
	if err != nil {
		return nil, err
	}
	
	lines := strings.Split(string(output), "\n")
	var rules []string
	for _, line := range lines {
		if line != "" && !strings.HasPrefix(line, "Chain") && !strings.HasPrefix(line, "num") {
			rules = append(rules, line)
		}
	}
	return rules, nil
}

func (m *IPTablesManager) BlockIP(ip string) error {
	return m.AddRule("INPUT", ip, "", "", "", "DROP")
}

func (m *IPTablesManager) AllowPort(port, protocol string) error {
	return m.AddRule("INPUT", "", "", protocol, port, "ACCEPT")
}

func main() {
	fw := NewIPTablesManager("filter")
	
	// Allow HTTP and HTTPS
	fw.AllowPort("80", "tcp")
	fw.AllowPort("443", "tcp")
	
	// Block malicious IP
	fw.BlockIP("10.0.0.100")
	
	// List rules
	rules, _ := fw.ListRules("INPUT")
	for _, rule := range rules {
		fmt.Println(rule)
	}
}
```

## 🌐 Web Application Firewall (WAF)

### WAF Rule Engine

```python
import re
from dataclasses import dataclass
from typing import List, Dict, Optional
from enum import Enum

class WAFAction(Enum):
    ALLOW = "allow"
    BLOCK = "block"
    LOG = "log"
    CHALLENGE = "challenge"

@dataclass
class WAFRule:
    id: str
    name: str
    pattern: str
    action: WAFAction
    severity: int  # 1-5
    description: str
    enabled: bool = True

class WAFEngine:
    def __init__(self):
        self.rules: List[WAFRule] = []
        self._load_default_rules()
    
    def _load_default_rules(self):
        """Load OWASP Core Rule Set patterns"""
        default_rules = [
            WAFRule(
                id="SQLI-001",
                name="SQL Injection Detection",
                pattern=r"(?i)(union.*select|select.*from|insert.*into|delete.*from|drop.*table|exec.*xp_|or\s+1\s*=\s*1)",
                action=WAFAction.BLOCK,
                severity=5,
                description="Detects common SQL injection patterns"
            ),
            WAFRule(
                id="XSS-001",
                name="XSS Detection",
                pattern=r"(?i)(<script|javascript:|onerror\s*=|onload\s*=|<iframe|<object|<embed)",
                action=WAFAction.BLOCK,
                severity=5,
                description="Detects cross-site scripting attempts"
            ),
            WAFRule(
                id="PATH-001",
                name="Path Traversal",
                pattern=r"(\.\.\/|\.\.\\|%2e%2e%2f|%2e%2e\/)",
                action=WAFAction.BLOCK,
                severity=4,
                description="Detects path traversal attempts"
            ),
            WAFRule(
                id="CMD-001",
                name="Command Injection",
                pattern=r"(;|\||`|\$\(|\$\{|\|\||&&)\s*(ls|cat|rm|wget|curl|nc|bash|sh|python|perl|ruby)",
                action=WAFAction.BLOCK,
                severity=5,
                description="Detects command injection patterns"
            ),
            WAFRule(
                id="SCANNER-001",
                name="Scanner Detection",
                pattern=r"(?i)(nikto|sqlmap|nmap|masscan|dirb|gobuster|wfuzz)",
                action=WAFAction.LOG,
                severity=3,
                description="Detects security scanner user agents"
            )
        ]
        self.rules.extend(default_rules)
    
    def add_custom_rule(self, rule: WAFRule):
        self.rules.append(rule)
    
    def inspect_request(self, request: Dict) -> Dict:
        """Inspect HTTP request for threats"""
        results = {
            "allowed": True,
            "violations": [],
            "risk_score": 0
        }
        
        # Combine all request parts for inspection
        inspect_strings = []
        for key in ["path", "query", "body", "headers", "cookies"]:
            if value := request.get(key):
                if isinstance(value, dict):
                    inspect_strings.extend(str(v) for v in value.values())
                else:
                    inspect_strings.append(str(value))
        
        combined_input = " ".join(inspect_strings)
        
        # Check against each rule
        for rule in self.rules:
            if not rule.enabled:
                continue
            
            if re.search(rule.pattern, combined_input):
                violation = {
                    "rule_id": rule.id,
                    "rule_name": rule.name,
                    "severity": rule.severity,
                    "action": rule.action.value,
                    "description": rule.description
                }
                results["violations"].append(violation)
                results["risk_score"] += rule.severity
                
                if rule.action == WAFAction.BLOCK:
                    results["allowed"] = False
        
        return results

# Usage
waf = WAFEngine()

request = {
    "path": "/api/users",
    "query": "id=1' OR '1'='1",
    "body": "",
    "headers": {"User-Agent": "Mozilla/5.0"},
    "cookies": {}
}

result = waf.inspect_request(request)
print(f"Request allowed: {result['allowed']}")
print(f"Risk score: {result['risk_score']}")
for v in result["violations"]:
    print(f"Violation: {v['rule_name']} (Severity: {v['severity']})")
```

## ☁️ Cloud Firewalls

### AWS Security Groups

```javascript
class AWSSecurityGroupBuilder {
    constructor(name, vpcId) {
        this.config = {
            GroupName: name,
            VpcId: vpcId,
            Description: '',
            InboundRules: [],
            OutboundRules: []
        };
    }
    
    description(desc) {
        this.config.Description = desc;
        return this;
    }
    
    allowInbound(port, protocol, source, description) {
        this.config.InboundRules.push({
            IpProtocol: protocol,
            FromPort: port,
            ToPort: port,
            IpRanges: [{ CidrIp: source, Description: description }]
        });
        return this;
    }
    
    allowHTTP(source = '0.0.0.0/0') {
        return this.allowInbound(80, 'tcp', source, 'HTTP');
    }
    
    allowHTTPS(source = '0.0.0.0/0') {
        return this.allowInbound(443, 'tcp', source, 'HTTPS');
    }
    
    allowSSH(source) {
        return this.allowInbound(22, 'tcp', source, 'SSH');
    }
    
    build() {
        return this.config;
    }
}

// Usage
const sg = new AWSSecurityGroupBuilder('web-server-sg', 'vpc-12345')
    .description('Security group for web servers')
    .allowHTTP()
    .allowHTTPS()
    .allowSSH('10.0.0.0/8')
    .build();

console.log(JSON.stringify(sg, null, 2));
```

## ❓ Interview Questions

### Basic
**Q: What's the difference between stateful and stateless firewalls?**
A: Stateless filters individual packets based on rules. Stateful tracks connection state and allows return traffic automatically.

**Q: Why is "default deny" important?**
A: Only explicitly allowed traffic passes, reducing attack surface. Errors in rules are safer (too restrictive vs too permissive).

### Intermediate
**Q: When would you use a WAF vs network firewall?**
A: Network firewall for L3/L4 filtering, network segmentation. WAF for L7 inspection, OWASP protection, application-specific attacks.

**Q: How do you handle firewall rules for a zero-trust network?**
A: Default deny all, micro-segmentation, identity-based rules, continuous verification, encrypted traffic between all segments.

### Advanced
**Q: Design firewall architecture for a multi-tier web application.**
A: (1) External firewall - allow 80/443 only, (2) DMZ - web servers, (3) Internal firewall - allow web tier to app tier on app port only, (4) App tier to database on DB port only, (5) Management network with jump host.
